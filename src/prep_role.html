%section% Prepositions of comparison

sui = "as" indefinite ......: sui + soft mutation definite ........: suin + nasal mutation
sui: Denethor bardha hui aran = "Denethor rules as [a] King" Noeg medir huin erain = "dwarves eat as Kings"
be = "as, like, according to" indefinite ......: be + soft mutation definite singular: ben + mixed mutation definite plural .: ben + nasal mutation
be: Denethor bardha ve aran = "Denethor rules as [a] King" Noeg medir ven erain = "dwarves eat as Kings"
