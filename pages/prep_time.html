%section% Prepositions of time

fo = "before" (temporal) ᴺS indefinite ......: fo + §s{soft mutation} definite singular: foe + §s{soft mutation} definite plural .: foen + §n{nasal mutation}
fo: Legolas ola §s{vara|bara} **fo** §s{vâd|mâd} = "Legolas becomes eager **before** [a] meal" Legolas ola §s{vara|bara} **foe** §s{vâd|mâd} = "Legolas becomes eager **before the** meal" Legolas ola §s{vara|bara} **foen** §n{vaid|maid} = "Legolas becomes eager **before the** meals"
ab = "after" indefinite ......: ab + §s{soft mutation} definite ........: ab + article
ab: Legolas ola §s{'elir|gelir} **ab** §s{vâd|mâd} = "Legolas becomes merry **after** [a] meal" Legolas olant §s{'elir|gelir} **ab** e·§s{vâd|mâd} = "Legolas became merry **after** the meal"
